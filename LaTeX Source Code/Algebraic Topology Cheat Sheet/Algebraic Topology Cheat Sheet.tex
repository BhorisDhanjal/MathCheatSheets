%------------ Constructions on topologies ---------------
\begin{tikzpicture}
	\node [mybox] (box){%
		\begin{minipage}{0.3\textwidth}
			\small
			Let \(X\) be a topological space.
			
			If \(Y\subseteq X\), the \textbf{subspace topology} on \(Y\) is
			\[
			\mathcal T_Y=\{U\cap Y:U\in \mathcal T_X\}.
			\]
			Equivalently, for the inclusion \(i:Y\hookrightarrow X\), a map \(h:Z\to Y\) is continuous iff \(i\circ h:Z\to X\) is continuous.
			\[
			\begin{tikzcd}[column sep=small,row sep=small]
				Z \arrow[r,"h"] \arrow[dr,"g=i\circ h"'] & Y \arrow[d,hook,"i"] \\
				& X
			\end{tikzcd}
			\]
			
			If \(q:X\twoheadrightarrow S\) is surjective, the \textbf{quotient topology} on \(S\) is defined by
			\[
			U\subseteq S \text{ open }\iff q^{-1}(U)\subseteq X \text{ open}.
			\]
			Equivalently, a map \(h:S\to Z\) is continuous iff \(h\circ q:X\to Z\) is continuous.
			\[
			\begin{tikzcd}[column sep=small,row sep=small]
				X \arrow[r,two heads,"q"] \arrow[dr,"g=h\circ q"'] & S \arrow[d,"h"] \\
				& Z
			\end{tikzcd}
			\]
			
			For a family \(\{X_a\}\), the \textbf{product topology} on \(\prod_a X_a\) is characterized by: \(h:Z\to \prod_a X_a\) is continuous iff each \(\pi_a\circ h:Z\to X_a\) is continuous.
			\[
			\begin{tikzcd}[column sep=small,row sep=small]
				& \prod_a X_a \arrow[d,"\pi_a"] \\
				Z \arrow[ur,"h"] \arrow[r,"g_a=\pi_a\circ h"'] & X_a
			\end{tikzcd}
			\]
			
			The \textbf{coproduct topology} on \(\coprod_a X_a\) is dual: \(h:\coprod_a X_a\to Z\) is continuous iff each \(h\circ \iota_a:X_a\to Z\) is continuous.
			\[
			\begin{tikzcd}[column sep=small,row sep=small]
				X_a \arrow[r,"\iota_a"] \arrow[dr,"h\circ \iota_a"'] & \coprod_a X_a \arrow[d,"h"] \\
				& Z
			\end{tikzcd}
			\]
		\end{minipage}
	};
	\node[fancytitle, right=10pt] at (box.north west) {Constructions on topologies};
\end{tikzpicture}


%------------ Deformation retraction ---------------
\begin{tikzpicture}
\node [mybox] (box){%
    \begin{minipage}{0.3\textwidth}
	  Let \(A\subseteq X\). A \textbf{deformation retraction} of \(X\) onto \(A\) is a homotopy \(H:X\times I\to X\) such that \(H(x,0)=x\), \(H(x,1)\in A\), and \(H(a,t)=a\) for all \(a\in A\), \(t\in I\). Equivalently, \(H_0=\mathrm{id}_X\), \(H_1=r\) for a retraction \(r:X\to A\), and \(H_t|_A=\mathrm{id}_A\) for all \(t\in I\).
  	
    \end{minipage}

%------------ Homotopy ---------------
\begin{tikzpicture}
	\node [mybox] (box){%
		\begin{minipage}{0.3\textwidth}
		Let \(f_0,f_1:X\to Y\) be continuous. A \textbf{homotopy} from \(f_0\) to \(f_1\) is a continuous map \(H:X\times I\to Y\) such that \(H(x,0)=f_0(x)\) and \(H(x,1)=f_1(x)\). Write \(f_0\simeq f_1\).
		
		Spaces \(X\) and \(Y\) are \textbf{homotopy equivalent} if there exist continuous maps \(f:X\to Y\) and \(g:Y\to X\) such that \(g\circ f\simeq \mathrm{id}_X\) and \(f\circ g\simeq \mathrm{id}_Y\).
		
		Homeomorphic \(\Rightarrow\) homotopy equivalent, but not conversely.
		
		A space \(X\) is \textbf{contractible} if \(\mathrm{id}_X\) is homotopic to a constant map.
		
		A map \(f:X\to Y\) is \textbf{null-homotopic} if \(f\simeq c\) for some constant map \(c:X\to Y\).
		\end{minipage}
	};
	%------------ Homotopy Header ---------------------
	\node[fancytitle, right=10pt] at (box.north west) {Homotopy};
\end{tikzpicture}


%------------ CW Complex ---------------
\begin{tikzpicture}
	\node [mybox] (box){%
		\begin{minipage}{0.3\textwidth}
			A \textbf{CW complex} is a space \(X=\bigcup_{n\ge 0}X^n\) such that \(X^0\) is a discrete set and, for each \(n\ge 1\), \(X^n\) is obtained from \(X^{n-1}\) by attaching \(n\)-cells \(e_\alpha^n\cong D^n\) along maps \(\varphi_\alpha:S^{n-1}\to X^{n-1}\). Equivalently, \(X^n\) is defined by the pushout
			\[
			\begin{tikzcd}[column sep=small,row sep=small]
				\coprod_\alpha S^{n-1}_\alpha & X^{n-1} \\
				\coprod_\alpha D^n_\alpha & X^n
				\arrow["{\coprod \varphi_\alpha}", from=1-1, to=1-2]
				\arrow[hook, from=1-1, to=2-1]
				\arrow[from=1-2, to=2-2]
				\arrow[from=2-1, to=2-2]
				\arrow["\lrcorner"{anchor=center, pos=0.125, rotate=180}, draw=none, from=2-2, to=1-1]
			\end{tikzcd}
			\]
			and \(X^n\setminus X^{n-1}=\coprod_\alpha e_\alpha^n\). The spaces \(X^n\) are the \(n\)-skeleta of \(X\).
		\end{minipage}
	};
	\node[fancytitle, right=10pt] at (box.north west) {CW Complex};
\end{tikzpicture}



%------------ Wedge, smash and suspension ---------------
\begin{tikzpicture}
	\node [mybox] (box){%
		\begin{minipage}{0.3\textwidth}
			For pointed spaces \((X,x_0)\), \((Y,y_0)\):
			
			\begin{itemize}
				\item \textbf{Wedge sum:} \(X\vee Y=(X\sqcup Y)/(x_0\sim y_0).\)
				\item \textbf{Smash product: }\( X\wedge Y=(X\times Y)/(X\vee Y).\)
				\item \textbf{Reduced suspension:} \( \Sigma X=(X\times I)/(X\times\{0\}\cup X\times\{1\}\cup \{x_0\}\times I). \)
			\end{itemize}
			If \(X,Y\) are pointed CW complexes, then \(X\vee Y\), \(X\wedge Y\), and \(\Sigma X\) admit CW structures.
		\end{minipage}
	};

%------------ Cone, cylinder and mapping cone ---------------
\begin{tikzpicture}
	\node [mybox] (box){%
		\begin{minipage}{0.3\textwidth}
			For a space \(X\):
			\begin{itemize}
				\item \textbf{Cone:} \(CX=(X\times I)/(X\times\{1\}).\)
				\item \textbf{Cylinder:} \(X\times I.\)
			\end{itemize}
			
			For \(f:X\to Y\):
			\begin{itemize}
				\item \textbf{Mapping cylinder:} \(M_f=((X\times I)\sqcup Y)/((x,0)\sim f(x)).\)
				\item \textbf{Mapping cone:} \(C_f=M_f/(X\times\{1\})\cong Y\cup_f CX.\)
			\end{itemize}
			
			If \(X,Y\) are CW complexes and \(f\) is cellular, then \(M_f\) and \(C_f\) admit CW structures.
		\end{minipage}

%------------ Paths and loops ---------------
\begin{tikzpicture}
	\node [mybox] (box){%
		\begin{minipage}{0.3\textwidth}
			Let \(I=[0,1]\).
			\begin{itemize}
				\item A \textbf{path} from \(x_0\) to \(x_1\) is a map \(\gamma:I\to X\) with \(\gamma(0)=x_0\), \(\gamma(1)=x_1\).
				\item A \textbf{loop} at \(x_0\) is a path \(\gamma:I\to X\) with \(\gamma(0)=\gamma(1)=x_0\).
				\item If \(\gamma:x_0\to x_1\), \(\eta:x_1\to x_2\), then \((\eta*\gamma)(t)=\gamma(2t)\) for \(0\le t\le \tfrac12\), and \((\eta*\gamma)(t)=\eta(2t-1)\) for \(\tfrac12\le t\le 1\).
				\item The reverse path is \(\overline{\gamma}(t)=\gamma(1-t)\).
				\item The loop space is \(\Omega(X,x_0)=\{\gamma:I\to X:\gamma(0)=\gamma(1)=x_0\}\).
			\end{itemize}
		\end{minipage}
	};
	%------------ Paths and loops Header ---------------------
	\node[fancytitle, right=10pt] at (box.north west) {Paths and loops};
\end{tikzpicture}

%------------ Fundamental groupoid ---------------
\begin{tikzpicture}
	\node [mybox] (box){%
		\begin{minipage}{0.3\textwidth}
			The \textbf{fundamental groupoid} \(\Pi_1(X)\) is the category with:
			\begin{itemize}
				\item objects: points \(x\in X\),
				\item morphisms \(x_0\to x_1\): paths \(\gamma:x_0\to x_1\) modulo homotopy rel endpoints.
			\end{itemize}
			Its composition is induced by concatenation, identities by constant paths, inverses by path reversal, so \(\Pi_1(X)\) is a \textbf{groupoid}.
			\begin{itemize}
				\item \(\Hom_{\Pi_1(X)}(x_0,x_1)=\{\text{paths }x_0\to x_1\}/\simeq \text{ rel }\partial I\).
				\item \(\Aut_{\Pi_1(X)}(x_0)=\Hom_{\Pi_1(X)}(x_0,x_0)\) is a group.
				\item A map \(f:X\to Y\) induces a functor \(f_*:\Pi_1(X)\to \Pi_1(Y)\), given by \(x\mapsto f(x)\) on objects and \([\gamma]\mapsto [f\circ \gamma]\) on morphisms.
				\item Thus \(\Pi_1:\mathbf{Top}\to \mathbf{Gpd}\) is a functor.
			\end{itemize}
		\end{minipage}
	};
	%------------ Fundamental groupoid Header ---------------------
	\node[fancytitle, right=10pt] at (box.north west) {Fundamental groupoid};
\end{tikzpicture}

%------------ Fundamental group ---------------
\begin{tikzpicture}
	\node [mybox] (box){%
		\begin{minipage}{0.3\textwidth}
			The \textbf{fundamental group} of \((X,x_0)\) is the automorphism group of \(x_0\) in \(\Pi_1(X)\):
			\begin{itemize}
				\item \(\pi_1(X,x_0)=\Aut_{\Pi_1(X)}(x_0)=\Hom_{\Pi_1(X)}(x_0,x_0)\),
				\item equivalently, \(\pi_1(X,x_0)=\{\text{loops at }x_0\}/\simeq \text{ rel }\partial I\).
			\end{itemize}
			The group structure is:
			\begin{itemize}
				\item product: \([\eta][\gamma]=[\eta*\gamma]\),
				\item identity: the constant loop class,
				\item inverse: \([\gamma]^{-1}=[\overline{\gamma}]\).
			\end{itemize}
			Functoriality for based spaces:
			\begin{itemize}
				\item a based map \(f:(X,x_0)\to (Y,y_0)\) induces \(f_*:\pi_1(X,x_0)\to \pi_1(Y,y_0)\), \([\gamma]\mapsto [f\circ\gamma]\),
				\item hence \(\pi_1:\mathbf{Top}_*\to \mathbf{Grp}\) is a functor.
				\item if \(\alpha:x_0\to x_1\) is a path, then \([\gamma]\mapsto [\overline{\alpha}*\gamma*\alpha]\) gives an isomorphism \(\pi_1(X,x_0)\cong \pi_1(X,x_1)\).
			\end{itemize}
		\end{minipage}
	};
	%------------ Fundamental group Header ---------------------
	\node[fancytitle, right=10pt] at (box.north west) {Fundamental group};
\end{tikzpicture}

%------------ van Kampen theorem for groupoids ---------------
\begin{tikzpicture}
\node [mybox] (box){%
	\begin{minipage}{0.3\textwidth}
		For \(A\subseteq X\), let \(\Pi_1(X,A)\) be the full subgroupoid of \(\Pi_1(X)\) on the objects \(A\).
		\begin{itemize}
			\item \(A\) is \textbf{representative} in \(X\) if it meets every path component of \(X\).
			\item Let \(X=U\cup V\) with \(U,V\) open, and assume \(A\subseteq U\cap V\) is representative in \(U\cap V\).
			\item Then \(A\) is representative in \(U\), \(V\), and \(X\).
			\item The inclusions induce a pushout square in \(\mathbf{Gpd}\):
		\end{itemize}
		\[
		\begin{tikzcd}[column sep=small,row sep=small]
			\Pi_1(U\cap V,A) & \Pi_1(U,A) \\
			\Pi_1(V,A) & \Pi_1(X,A)
			\arrow[from=1-1, to=1-2]
			\arrow[from=1-1, to=2-1]
			\arrow[from=1-2, to=2-2]
			\arrow[from=2-1, to=2-2]
			\arrow["\lrcorner"{anchor=center, pos=0.125, rotate=180}, draw=none, from=2-2, to=1-1]
		\end{tikzcd}
		\]
		\begin{itemize}
			\item For \(A=\{x_0\}\) and connected \(U,V,U\cap V\), this recovers the usual group version.
			\item This is a special case of the stronger fact that \(\Pi_1\) preserves homotopy colimits.
		\end{itemize}
	\end{minipage}
};

\end{tikzpicture}
%------------ Covering spaces ---------------
\begin{tikzpicture}
	\node [mybox] (box){%
		\begin{minipage}{0.3\textwidth}
			A \textbf{covering map} is a map \(p:\widetilde X\to X\) such that every \(x\in X\) has an open neighborhood \(U\) with \(p^{-1}(U)=\bigsqcup_\alpha U_\alpha\), where each \(p|_{U_\alpha}:U_\alpha\to U\) is a homeomorphism.
			\begin{itemize}
				\item Such a \(U\) is \textbf{evenly covered}.
				\item A \textbf{morphism of coverings} \(g:(\widetilde X,p)\to (\widehat X,q)\) is a map with \(q\circ g=p\).
				\item Let \(\mathrm{Cov}(X)\) denote the category of coverings over \(X\).
				\item A covering is a local homeomorphism.
			\end{itemize}
			\[
			\begin{tikzcd}
				\widetilde X \arrow[rr,"g"] \arrow[dr,"p"'] && \widehat X \arrow[dl,"q"]\\
				& X &
			\end{tikzcd}
			\qquad
			\begin{tikzcd}
				U_\alpha \arrow[r,"p|_{U_\alpha}","\cong"'] \arrow[d,hook]
				& U \arrow[d,hook] \\
				\widetilde X \arrow[r,"p"']
				& X
			\end{tikzcd}
			\]
		\end{minipage}
	};
	%------------ Covering spaces Header ---------------------
	\node[fancytitle, right=10pt] at (box.north west) {Covering spaces};
\end{tikzpicture}

%------------ Path lifting and homotopy lifting ---------------
\begin{tikzpicture}
	\node [mybox] (box){%
		\begin{minipage}{0.3\textwidth}
			Lifting properties of a covering \(p:\widetilde X\to X\).
			\begin{itemize}
				\item \textbf{Unique path lifting:} if \(\gamma:I\to X\) and \(\widetilde x_0\in p^{-1}(\gamma(0))\), there is a unique lift \(\widetilde\gamma:I\to \widetilde X\) with \(\widetilde\gamma(0)=\widetilde x_0\).
				\item \textbf{Unique homotopy lifting:} if \(H:Y\times I\to X\) and \(\widetilde H_0:Y\to \widetilde X\) lifts \(H|_{Y\times\{0\}}\), then there is a unique lift \(\widetilde H:Y\times I\to \widetilde X\) with \(\widetilde H|_{Y\times\{0\}}=\widetilde H_0\).
				\item If \(\widetilde X\) is path connected, then \(p_*:\pi_1(\widetilde X,\widetilde x_0)\to \pi_1(X,x_0)\) is injective.
				\item Fiber transport along a path \(\gamma:x\to y\) gives a bijection \(T_\gamma:p^{-1}(x)\to p^{-1}(y)\).
			\end{itemize}
			\[
			\begin{tikzcd}
				& \widetilde X \arrow[d,"p"] \\
				I \arrow[ur,dashed,"\widetilde\gamma"] \arrow[r,"\gamma"']
				& X
			\end{tikzcd}
			\qquad
			\begin{tikzcd}
				Y\times\{0\} \arrow[r,"\widetilde H_0"] \arrow[d,hook]
				& \widetilde X \arrow[d,"p"] \\
				Y\times I \arrow[ur,dashed,"\widetilde H"] \arrow[r,"H"']
				& X
			\end{tikzcd}
			\]
		\end{minipage}

%------------ Classification of covering spaces ---------------
\begin{tikzpicture}

%------------ Group actions, orbit spaces, and Galois correspondence ---------------
\begin{tikzpicture}
	\node [mybox] (box){%
		\begin{minipage}{0.3\textwidth}
			Let \(p:\widetilde X\to X\) be a universal cover and identify \(G=\mathrm{Deck}(\widetilde X/X)\cong \pi_1(X,x_0)\).
			\begin{itemize}
				\item For each subgroup \(H\le G\), the quotient \(\widetilde X/H\to X\) is a connected covering.
				\item Every connected covering of \(X\) is isomorphic to \(\widetilde X/H\) for some \(H\le G\).
				\item If \(H\le K\le G\), there is a covering \(\widetilde X/H\to \widetilde X/K\) over \(X\).

				\item In general \(\mathrm{Deck}(\widetilde X/H\,/\,X)\cong N(H)/H\), and if \(H\trianglelefteq G\), then \(\mathrm{Deck}(\widetilde X/H\,/\,X)\cong G/H\).
			\end{itemize}
		\end{minipage}

%------------ Chain complexes and exactness ---------------
\begin{tikzpicture}
	\node [mybox] (box){%
		\begin{minipage}{0.3\textwidth}
			Fix a ring \(R\). A \textbf{chain complex} \(C_*\) of \(R\)-modules is a sequence of maps
			\(\cdots \to C_{n+1}\xrightarrow{\partial_{n+1}} C_n\xrightarrow{\partial_n} C_{n-1}\to \cdots\)
			with \(\partial_n\partial_{n+1}=0\) for all \(n\).
			\begin{itemize}
				\item \(Z_n(C_*)=\ker(\partial_n)\) are the \(n\)-cycles.
				\item \(B_n(C_*)=\mathrm{im}(\partial_{n+1})\) are the \(n\)-boundaries.
				\item Since \(\partial_n\partial_{n+1}=0\), one has \(B_n(C_*)\subseteq Z_n(C_*)\).
				\item \(H_n(C_*)=Z_n(C_*)/B_n(C_*)=\ker(\partial_n)/\mathrm{im}(\partial_{n+1})\).
				\item A homogeneous map \(f:C_*\to D_*\) has \textbf{degree} \(r\) if \(f_n:C_n\to D_{n+r}\) for all \(n\).
				\item A \textbf{chain map} is a degree \(0\) map \(f:C_*\to D_*\) with \(\partial f=f\partial\), hence induces \(f_*:H_n(C_*)\to H_n(D_*)\).
				\item A \textbf{chain homotopy} \(h:f\simeq g\) between chain maps \(f,g:C_*\to D_*\) is a degree \(1\) map \(h:C_*\to D_*\) with \(f-g=\partial h+h\partial\).
				\item \(C_*\) is \textbf{exact} if \(\mathrm{im}(\partial_{n+1})=\ker(\partial_n)\) for all \(n\), equivalently \(H_n(C_*)=0\) for all \(n\).
			\end{itemize}
		\end{minipage}
	};
	%------------ Chain complexes and exactness Header ---------------------
	\node[fancytitle, right=10pt] at (box.north west) {Chain complexes and exactness};
\end{tikzpicture}

%------------ Free, projective, and injective modules ---------------
\begin{tikzpicture}
	\node [mybox] (box){%
		\begin{minipage}{0.3\textwidth}
			Fix a ring \(R\).
			\begin{itemize}
				\item An \(R\)-module \(F\) is \textbf{free} if \(F\cong \bigoplus_{i\in I} R\).
				\item An \(R\)-module \(P\) is \textbf{projective} if for every epimorphism \(g:M\twoheadrightarrow N\) and map \(f:P\to N\), there exists \(\widetilde f:P\to M\) with \(g\widetilde f=f\).
				\item Equivalently, \(P\) is projective iff \(P\) is a direct summand of a free module.
				\item An \(R\)-module \(I\) is \textbf{injective} if for every monomorphism \(j:M\hookrightarrow N\) and map \(f:M\to I\), there exists \(\widetilde f:N\to I\) with \(\widetilde f j=f\).
				\item Every free module is projective.
				\item In \(R\)-Mod, there are enough projectives and enough injectives.
			\end{itemize}
		\end{minipage}
	};
	%------------ Free, projective, and injective modules Header ---------------------
	\node[fancytitle, right=10pt] at (box.north west) {Free, projective, and injective modules};
\end{tikzpicture}

%------------ Resolutions ---------------
\begin{tikzpicture}
	\node [mybox] (box){%
		\begin{minipage}{0.3\textwidth}
			Let \(M\) be an \(R\)-module.
			\begin{itemize}
				\item A \textbf{free resolution} of \(M\) is an exact sequence \(\cdots \to F_2\to F_1\to F_0\to M\to 0\) with each \(F_i\) free.
				\item A \textbf{projective resolution} of \(M\) is an exact sequence \(\cdots \to P_2\to P_1\to P_0\to M\to 0\) with each \(P_i\) projective.
				\item An \textbf{injective resolution} of \(M\) is an exact sequence \(0\to M\to I^0\to I^1\to I^2\to \cdots\) with each \(I^i\) injective.
				\item Any two projective resolutions of \(M\) are chain homotopy equivalent.
				\item Any two injective resolutions of \(M\) are chain homotopy equivalent.
				\item If \(R=\mathbb Z\), every Abelian group \(M\) has a free resolution \(0\to F_1\to F_0\to M\to 0\), since \(\mathbb Z\) is a PID and submodules of free modules are free.
			\end{itemize}
		\end{minipage}
	};
	%------------ Resolutions Header ---------------------
	\node[fancytitle, right=10pt] at (box.north west) {Resolutions};
\end{tikzpicture}
%------------ Exact functors and derived functors ---------------
\begin{tikzpicture}
	\node [mybox] (box){%
		\begin{minipage}{0.3\textwidth}
			Let \(F:\mathcal A\to\mathcal B\) be an additive covariant functor, with \(\mathcal A\) abelian.
			\begin{itemize}
				\item \(F\) is \textbf{left exact} if \(0\to A'\to A\to A''\) exact implies \(0\to F(A')\to F(A)\to F(A'')\) exact.
				\item \(F\) is \textbf{right exact} if \(A'\to A\to A''\to 0\) exact implies \(F(A')\to F(A)\to F(A'')\to 0\) exact.
				\item Assume \(\mathcal A\) has enough projectives and \(F\) is right exact. For \(A\in\mathcal A\), choose a projective resolution \(\cdots \to P_2\xrightarrow{d_2}P_1\xrightarrow{d_1}P_0\to A\to 0\).
				\item Applying \(F\) degreewise gives a chain complex \(F(P_*)\), since \(F(d_i)F(d_{i+1})=F(d_id_{i+1})=0\).
				\item The \textbf{left derived functors} of \(F\) are \(L_iF(A)=H_i(F(P_*))\). In particular \(L_0F\cong F\).
				\item Dually, if \(\mathcal A\) has enough injectives and \(F\) is left exact, then for an injective resolution \(0\to A\to I^0\xrightarrow{d^0}I^1\xrightarrow{d^1}\cdots\), the \textbf{right derived functors} are \(R^iF(A)=H^i(F(I^*))\). In particular \(R^0F\cong F\).
				\item These are well-defined up to natural isomorphism since projective and injective resolutions are unique up to chain homotopy.
				\item A short exact sequence \(0\to A'\to A\to A''\to 0\) induces natural connecting morphisms and hence long exact sequences of left or right derived functors.
			\end{itemize}
		\end{minipage}
	};
	%------------ Exact functors and derived functors Header ---------------------
	\node[fancytitle, right=10pt] at (box.north west) {Exact functors and derived functors};
\end{tikzpicture}

%------------ Tensor, Hom, Tor, Ext ---------------
\begin{tikzpicture}
	\node [mybox] (box){%
		\begin{minipage}{0.3\textwidth}
			Let \(R\) be a ring.
			\begin{itemize}
				\item If \(M\) is a right \(R\)-module, then \(M\otimes_R -\) is covariant right exact.
				\item If \(N\) is a left \(R\)-module, then \(-\otimes_R N\) is covariant right exact.
				\item If \(M\) is a left \(R\)-module, then \(\mathrm{Hom}_R(M,-)\) is covariant left exact.
				\item \(\mathrm{Tor}_i^R(M,N)=L_i(M\otimes_R -)(N)\).
				\item \(\mathrm{Ext}_R^i(M,N)=R^i\mathrm{Hom}_R(M,-)(N)\).
				\item For \(P_*\to M\to 0\) projective resolution, \(\mathrm{Tor}_i^R(M,N):=H_i(P_*\otimes_R N)\).
				\item For \(0\to N\to I^\bullet\) injective resolution,  \(\mathrm{Ext}_R^i(M,N):=H^i(\mathrm{Hom}_R(M,I^\bullet))\).
				\item Over \(\mathbb Z\): \(\mathrm{Tor}_1^{\mathbb Z}(\mathbb Z/n,A)\cong \{a\in A:na=0\}\), \(\mathrm{Ext}_{\mathbb Z}^1(\mathbb Z/n,A)\cong A/nA\).
			\end{itemize}
		\end{minipage}
	};

\end{tikzpicture}

%------------ Kunneth theorem ---------------
\begin{tikzpicture}
